JavaScript

  1. XMLHttpRequest 获取服务器 json 文件内容
// playlist
let requestURL = 'json/songs.json';
let request = new XMLHttpRequest();

var songsList;
var musicPlayer;
var musicImages;
var scount = 0;

// XMLHttpRequestEventTarget.onload 是 XMLHttpRequest 请求成功完成时调用的函数
request.onload = function(e) {
    var files2 = request.responseText;

    var json = JSON.parse(files2);
    var listLength = json["data"].length;

    songsList = new Array();
    for (var i = 0; i < listLength; i++) {
        var imagesMax = json["data"][i]["album"]["images"].length - 1;

        songsList[i] = [json["data"][i]["name"], json["data"][i]["album"]["images"][imagesMax != 0 ? imagesMax - 1 : imagesMax]["url"]];
    }

    shuffle(songsList);

    playMusic("#music-player", "music-img");
}

request.open('GET', requestURL);
request.responseType = 'text';
request.send();

// shuffle the list
function shuffle(a) {
    for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
    }
    return a;
}

// function to move to next file
function next() {
    // set next song's src
    (scount === (songsList.length - 1)) ? (scount = 0) : scount++;
    musicPlayer.src = "mus/" + songsList[scount][0] + ".mp3";

    // plan 2
    if (musicImages[0].style.opacity == "1") {
        musicImages[0].style.opacity = 0;
        musicImages[0].style.left = "-30px";
        musicImages[1].src = songsList[scount][1];
        musicImages[1].style.opacity = 1;
        musicImages[1].style.left = "0px";
    } else {
        musicImages[1].style.opacity = 0;
        musicImages[1].style.left = "-30px";
        musicImages[0].src = songsList[scount][1];
        musicImages[0].style.opacity = 1;
        musicImages[0].style.left = "0px";
    }

    // set transitionend
    musicImages[0].addEventListener("transitionend", changeTrans, false);

    musicPlayer.load();
    musicPlayer.play();
}

// function of transitionend
function changeTrans() {
    if (musicImages[0].style.opacity == "0") {
        musicImages[0].src = songsList[scount][1];
        musicImages[0].style.left = "30px";
    } else {
        musicImages[1].src = songsList[scount][1];
        musicImages[1].style.left = "30px";
    }
}

// function of audio control bind div
function audioControl() {
    if (musicPlayer.paused) {
        musicPlayer.play();
        // 兼容IE8以下
        // document.styleSheets[0].addRule('#image-box::before','display: none');
        // 兼容现代浏览器
        document.styleSheets[0].insertRule('#image-box::before { visibility: hidden }', 0);
    } else {
        musicPlayer.pause();
        document.styleSheets[0].deleteRule(0);
    }
}
