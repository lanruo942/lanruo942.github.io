问题

  • strapi yarn install 报错，better-sqlite3 构建错误。
    • 解决：yarn add better-sqlite3
  • pm2 配置（yarn global add pm2）
    • 启动：在项目目录下执行 pm2 start yarn --name "project-name" -- start
    • 查看状态：pm2 list
    • 保存状态：pm2 save
    • 自动重启：pm2 startup
  • nginx 配置：
    • nextjs 配置
server {
	listen 443 ssl http2;
	listen [::]:442 ssl http2;
	server_name <your_domain>;

	ssl_certificate /usr/local/etc/nextjs/your_cert.crt;
	ssl_certificate_key /usr/local/etc/nextjs/your_private.key;

	location / {
		# reverse proxy for next server
		proxy_pass http://localhost:3000;
		proxy_http_version 1.1;
		proxy_set_header Host $host;
		proxy_set_header Connection 'upgrade';
		proxy_set_header upgrade $http_upgrade;
		proxy_cache_bypass $http_upgrade;

		# we need to remove this 404 handling
		# because next's _next folder and own handling
		# try_files $uri $uri/ =404;
	}

	location ~ /.well-known {
		allow all;
	}
}

server {
	listen 80;
	server_name <your_domain>;

	return 301 https://$host$request_uri;

	location / {
		# this must stay localhost. The port must be the same as your Next.js project
		proxy_pass http://localhost:3000;
		proxy_http_version 1.1;
		proxy_set_header Upgrade $http_upgrade;
		proxy_set_header Connection 'upgrade';
		proxy_set_header Host $host;
		proxy_cache_bypass $http_upgrade;
	}

	# simple headers
	add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
	add_header X-XSS-Protection          "1; mode=block" always;
	add_header X-Frame-Options DENY always;
}

- strapi 配置

# Strapi server
upstream <your_domain> {
	server 127.0.0.1:1337;
}

server {
	listen 443 ssl http2;
	listen [::]:442 ssl http2;
	server_name <your_domain>;

	ssl_certificate /usr/local/etc/strapi/your_cert.crt;
	ssl_certificate_key /usr/local/etc/strapi/your_private.key;

	location / {
		# reverse proxy for next server
		proxy_pass http://localhost:1337;
		proxy_http_version 1.1;
		proxy_set_header X-Forwarded-Host $host;
		proxy_set_header X-Forwarded-Server $host;
		proxy_set_header X-Real-IP $remote_addr;
		proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
		proxy_set_header X-Forwarded-Proto $scheme;
		proxy_set_header Host $http_host;
		proxy_set_header Upgrade $http_upgrade;
		proxy_set_header Connection "Upgrade";
		proxy_pass_request_headers on;
	}
}

server {
	listen 80;
	server_name <your_domain>;

	return 301 https://$host$request_uri;

	location / {
		proxy_pass http://localhost:1337;
	}
}
  • strapi 目录下的 config/server.js 添加 url：
require('dotenv').config();

module.exports = ({ env }) => ({
	host: env('HOST', process.env.HOST || '0.0.0.0'),
	port: env.int('PORT', process.env.PORT || 1337),
	app: {
		keys: env.array('APP_KEYS', process.env.APP_KEYS),
	},
	url: 'https://<your_domain>'    // this line
});
  • 如果出现 502 bad Gateway，一般是 pm2 的问题，删除列表中出问题的进程并重新启动。或者 stop 后再 start 尝试修复。
  • 80 端口 return 重定向后的内容其实没什么用，不过为了在申请证书前 80 端口能正常使用，还是保留这些内容。
